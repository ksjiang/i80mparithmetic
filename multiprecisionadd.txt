***8085 MULTIPRECISION ADDITION ROUTINE***
***Result memory block must be cleared for proper operation.***

ADD0:	LXI B, X
	LXI H, Z
	MVI D, SL
ADD1:	LDAX B		;copy addend to result block
	MOV M, A
	INX B
	INX H
	DCR D
	JNZ ADD1
ADD2:	LXI B, Y
	LXI H, Z
	MVI D, SL
	ORA A		;reset carry flag
ADD3:	LDAX B		;main addition loop
	ADC M
	MOV M, A
	DCR D
	JZ ADD4
	INX B
	INX H
	JMP ADD3
ADD4:	JNC ADD5
	INX H
	INR M
ADD5:	HLT

SL	EQU	03H
X	EQU	0FE00H
Y	EQU	0FE80H
Z	EQU	0FF00H
